1 LeetCode 2 两数相加

1.1 题目分析

高精度加法，链表实现，基础题目。

1.2 Java代码

class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
	ListNode result = new ListNode(0);
	ListNode now = result;
	ListNode pre = null;
	while(l1 != null || l2 != null) {
	    if(l1 != null) {
		now.val += l1.val;
		l1 = l1.next;
	    }
	    if(l2 != null) {
		now.val += l2.val;
		l2 = l2.next;
	    }
	    now.next = new ListNode(now.val/10);
	    now.val %= 10;
	    pre = now;
	    now = now.next;
	}
	if(now.val == 0) {
	    pre.next = null;
	}
	return result;
    }
}
