1.2 Java代码

class Solution {
    public int[] twoSum(int[] nums, int target) {
	int[] result = {0, 0};
	for(int i=0; i<nums.length; i++) {
	    for(int j=i+1; j<nums.length; j++) {
		if(nums[i] + nums[j] == target) {
		    result[0] = i;
		    result[1] = j;
		    return result;  // comment for test2
		}
	    }
	}
    }
}
